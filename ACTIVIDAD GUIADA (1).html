import os
import matplotlib.pyplot as plt
import numpy as np

#Se obtienen y grafican las curvas de luz de las estrellas cefidas
for starfile in os.listdir('./dataHST'):
    time = []
    magnitude = []
    
    for line in open('./dataHST/'+starfile, 'r'):
        lines = [i for i in line.split()]
        time.append(float(lines[0]))
        magnitude.append(float(lines[1]))
        
    plt.figure(figsize=(10,5))
    plt.title(starfile[0:len(starfile)-4])
    plt.xlabel('Tiempo (dias)')
    plt.ylabel('Magnitud aparente')
    plt.yticks(np.arange(round(min(magnitude),1), round(max(magnitude),1), step=0.1))
    plt.gca().invert_yaxis()
    plt.plot(time, magnitude, marker = 'o', c = 'g')    
    plt.savefig(starfile[0:len(starfile)-4]+'.png')
    plt.show()
